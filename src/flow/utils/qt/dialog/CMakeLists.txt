# -------------------------------------------------------------------------------------------------- #
# -------------------------------------- Set CMake Version ----------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
cmake_minimum_required(VERSION 3.11)
# -------------------------------------------------------------------------------------------------- #
# -------------------------------------- Set files variable ---------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
set(source_root ${FLOW_SOURCE_DIR}/src/flow/utils/qt/dialog)
set(include_root ${FLOW_SOURCE_DIR}/include/flow/utils/qt/dialog)

set(sources
        ${source_root}/dialog_with_toggle_view.cpp
        ${source_root}/extended_file_dialog.cpp)

set(includes
        ${include_root}/export.h
        ${include_root}/dialog_with_url_links.h
        ${include_root}/dialog_with_toggle_view.h
        ${include_root}/extended_file_dialog.h)
# -------------------------------------------------------------------------------------------------- #
# ---------------------------------- Qt libraries configuration ------------------------------------ #
# -------------------------------------------------------------------------------------------------- #
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)
# -------------------------------------------------------------------------------------------------- #
# ----------------------------------- Find external libraries -------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
find_package(Qt6 REQUIRED COMPONENTS Widgets)
# -------------------------------------------------------------------------------------------------- #
# ------------------------------------ Create utils shared lib ------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
flow_add_utils(dialog
        SOURCES ${sources} ${includes}
        DEPENDS Qt6::Widgets)
# -------------------------------------------------------------------------------------------------- #
